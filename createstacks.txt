Create Fargate Docker Cluster
#############################
aws cloudformation create-stack --stack-name ecsteststack --template-body file://cloudformation/cluster.yml --capabilities CAPABILITY_IAM

Create Fargate Service with Pipline
###################################
aws cloudformation create-stack --stack-name ecsserviceteststack --template-body file://cloudformation/servicev2.yml --capabilities CAPABILITY_IAM \
--parameters ParameterKey=StackName,ParameterValue=ecsteststack \
ParameterKey=GitHubUser,ParameterValue="seankbeard" \
ParameterKey=GitHubRepo,ParameterValue="docker-sample" \
ParameterKey=GitHubBranch,ParameterValue="dev-fork" \
ParameterKey=GitHubToken,ParameterValue="" \
ParameterKey=ServiceName,ParameterValue=Webdemo-Service \
ParameterKey=ImageUrl,ParameterValue=194783185645.dkr.ecr.ap-southeast-2.amazonaws.com/sean-test \
ParameterKey=Path,ParameterValue="/webdemo/*" \
ParameterKey=PathHealthCheck,ParameterValue="/webdemo/index.html" \
ParameterKey=Priority,ParameterValue=1 \
ParameterKey=ContainerPort,ParameterValue=80 \
ParameterKey=DesiredCount,ParameterValue=2
